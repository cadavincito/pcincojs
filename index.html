        function mousePressed(){
          let p = new Particula();
          particulas.push(p);
        }
        
        function mouseDragged(){
          let p = new Particula();
          particulas.push(p);
        }
        
        function moverYmostrar(p){
          p.mover();
          p.mostrar();
        }
        
        class Particula {
          constructor(){
            this.x = mouseX;
            this.y = mouseY;
            this.vx = random(-5,5);
            this.vy = random(-5,5);
            this.size = random(20,50);
            this.color = random(colors);
            this.forma = 'circulo';
          }
          mover(){
            this.x = this.x + this.vx;
            this.y = this.y + this.vy
            
            if (this.x>=width){
              this.vx = this.vx * -1;
              this.forma = 'cuadrado'
            }
            
            if (this.x<=0){
              this.vx = this.vx * -1;
              this.forma = 'triangulo'
            }
            
            if (this.y>=height){
              this.vy = this.vy * -1;
              this.forma = 'circulo'
            }
            
            if (this.y<=0){
              this.vy = this.vy * -1;
              this.forma = 'circulo'
            } 
            
          }
          mostrar(){
            noStroke();
            fill(this.color)
            
            if (this.forma=='circulo'){
              circle(this.x,this.y,this.size);
            }
            if (this.forma == 'cuadrado'){
              square(this.x,this.y,this.size);
            }
            if (this.forma == 'triangulo'){
              triangle(this.x+this.size/2, this.y + this.size/2, this.x - this.size / 2, this.y + this.size/2,this.x,this.y-this.size/2);
            }
          }
        }

        function setup() {
          createCanvas(400, 400);
          background(0); // Set background color to black
          noStroke();
          drawKandinsky();
      }
      
      function drawKandinsky() {
          let colors = ['#FF5733', '#33FF57', '#FFFFFF', '#FF57F7', '#FFF333', '#FF0020'];
          let shapes = ['circle', 'triangle'];
          let x, y, size;
          let numShapes = 20;
      
          for (let i = 0; i < numShapes; i++) {
              let shape = random(shapes);
              x = random(width);
              y = random(height);
              size = random(40, 100);
      
              fill(random(colors));
              if (shape === 'circle') {
                  ellipse(x, y, size, size);
              } else if (shape === 'triangle') {
                  let x2 = x + random(-200, 200);
                  let y2 = y + random(-200, 200);
                  let x3 = x + random(-200, 200);
                  let y3 = y + random(-200, 200);
                  triangle(x, y, x2, y2, x3, y3);
              }
          }
      }
