        class Particula {
          constructor(){
            this.x = mouseX;
            this.y = mouseY;
            this.vx = random(-5,5);
            this.vy = random(-5,5);
            this.size = random(20,50);
            this.color = random(colors);
            this.forma = 'circulo';
          }
          mover(){
            this.x = this.x + this.vx;
            this.y = this.y + this.vy
            
            if (this.x>=width){
              this.vx = this.vx * -1;
              this.forma = 'cuadrado'
            }
            
            if (this.x<=0){
              this.vx = this.vx * -1;
              this.forma = 'triangulo'
            }
            
            if (this.y>=height){
              this.vy = this.vy * -1;
              this.forma = 'circulo'
            }
            
            if (this.y<=0){
              this.vy = this.vy * -1;
              this.forma = 'circulo'
            } 
            
          }
          mostrar(){
            noStroke();
            fill(this.color)
            
            if (this.forma=='circulo'){
              circle(this.x,this.y,this.size);
            }
            if (this.forma == 'cuadrado'){
              square(this.x,this.y,this.size);
            }
            if (this.forma == 'triangulo'){
              triangle(this.x+this.size/2, this.y + this.size/2, this.x - this.size / 2, this.y + this.size/2,this.x,this.y-this.size/2);
            }
          }
        }

class Particula {
  constructor(){
    this.x = random(width)
    this.y = random(height)
    this.vx = random(-2,2)
    this.vy = random(-2, 2)
    this.col = random(colores)
    this.s = random(10,60)
    this.startTime= random(0, 200)
  }
  dibujar() {
    fill(this.col)
    drawingContext.shadowColor = this.col
    strokeWeight(2)
    circle(this.x, this.y, this.s * sin(( frameCount + this.startTime) *0.01))
  }
  mover(){
    this.x = this.x + this.vx
    this.y = this.y + this.vy
    
  if (this.x > width) {
    this.x = 0
  } 
  if (this.x < 0) {
    this.x = width
    
    if (this.y > width) {
    this.y = 0
  } 
  if (this.y < 0) {
    this.y = width
  }
    
  }
}

        function setup() {
          createCanvas(400, 400);
          background(0); // Set background color to black
          noStroke();

      function drawKandinsky() {
          let colors = ['#FF5733', '#33FF57', '#FFFFFF', '#FF57F7', '#FFF333', '#FF0020'];
          let shapes = ['circle', 'triangle'];
          let x, y, size;
          let numShapes = 20;
      
          for (let i = 0; i < numShapes; i++) {
              let shape = random(shapes);
              x = random(width);
              y = random(height);
              size = random(40, 100);
      
              fill(random(colors));
              if (shape === 'circle') {
                  ellipse(x, y, size, size);
              } else if (shape === 'triangle') {
                  let x2 = x + random(-200, 200);
                  let y2 = y + random(-200, 200);
                  let x3 = x + random(-200, 200);
                  let y3 = y + random(-200, 200);
                  triangle(x, y, x2, y2, x3, y3);
              }
          }
      }

Bouncing Balls



 
        let balls = [];

        function setup() {
          createCanvas(600, 400);
          createBalls();
        }
        
        function draw() {
          
          background(0);
        
          for (let ball of balls) {
            ball.move();
            ball.displayBall();
            ball.checkEdge();
          }
        }
        
        function createBalls(){
            
          for (let i = 0; i < 100; i++) {
            let ball = new Ball(random(width), random(height), random(10, 30), color(random(255), random(255), random(255)));
            balls.push(ball);
          }
        }
        
        class Ball {
          constructor(x, y, diameter, col) {
            this.x = x;
            this.y = y;
            this.diameter = diameter;
            this.col = col;
            this.xSpeed = random(-2, 2);
            this.ySpeed = random(-2, 2);
          }
        
          move() {
            this.x += this.xSpeed;
            this.y += this.ySpeed;
          }
        
          displayBall() {
            fill(this.col);
            ellipse(this.x, this.y, this.diameter, this.diameter);
          }
        
          checkEdge() {
            if (this.x < 0 || this.x > width) {
              this.xSpeed *= -1;
            }
            if (this.y < 0 || this.y > height) {
              this.ySpeed *= -1;
            }
          }
        }
